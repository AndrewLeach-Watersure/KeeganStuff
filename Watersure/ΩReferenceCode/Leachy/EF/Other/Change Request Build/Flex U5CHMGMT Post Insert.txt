DECLARE
@v_id integer,

@wonum nvarchar(8),
@wodesc nvarchar(80),
@desc nvarchar(80),
@changecheck nvarchar(1),
@riskRating nvarchar(10);

EXECUTE @v_id = O7SESS_CUR_USER @user OUTPUT;
BEGIN
    SELECT @wonum = CMF_EVENT,
                   @wodesc = EVT_DESC, 
                   @status = EVT_STATUS,
                   @type = EVT_TYPE,
                   @changecheck = EVT_UDFCHKBOX03,
                   @riskRating = CMF_RISKTOTAL,
                   @desc = CMF_DESCRIPTION

                 FROM U5CHMGMT
LEFT OUTER JOIN R5EVENTS ON CMF_EVENT = EVT_CODE
                 WHERE  SQLIDENTITY = :rowid

IF @desc is null 
BEGIN
SET @desc = @wodesc;
END
IF @changecheck = '-' OR @changecheck IS NULL
BEGIN
RAISERROR('You can only fill out a change request for a Change type work order  ''%s''',16,1,@wonum);
END
BEGIN
IF @riskRating <= 0 RAISERROR('Risk Rating cannot be 0',16,1)
IF @riskRating <= 4 SET @date = null
IF (@riskRating <= 9 AND @riskRating > 4) SET @date = DATEADD(month, 1, GETDATE())
IF (@riskRating <= 14 AND @riskRating > 9) SET @date = DATEADD(week, 2, GETDATE())
IF (@riskRating <= 25 AND @riskRating > 14) SET @date = DATEADD(week, 2, GETDATE())
IF @riskRating > 25 RAISERROR('Risk Rating cannot be above 25',16,1)
END
BEGIN
IF TRIGGER_NESTLEVEL() > 4
RETURN
IF @changecheck = '+'
UPDATE U5CHMGMT
SET DESCRIPTION = @wodesc,
CMF_EVTSTATUS = @status,
CMF_EVTTYPE = dbo.repgetdesc('EN','UCOD',@type, 'JBTP',null),
